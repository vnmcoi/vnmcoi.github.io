**Vấn đề:** Vì chúng ta có các thao tác loại $1$ yêu cầu cập nhật nên nếu chúng ta sử dụng cập nhật lười của mảng cộng dồn thì thuật toán của chúng ta sẽ có độ phức tạp là $\mathcal{O}(N \times Q)$.

Vậy chúng ta cần phải có một cách làm khác đó là sử dụng các cấu trúc dữ liệu như IT hoặc BIT.

Lời giải bên dưới là sử dụng BIT (Fenwick Tree).

```
#include <bits/stdc++.h>
using namespace std;

const int mxN = 5e5 + 5;

int n, q;
int x[mxN];

struct FenwickTree
{
    int n;
    vector<long long> bit;

    FenwickTree(int _n)
    {
        n = _n;
        if (n != 0)
        {
            bit.assign(n + 1, 0);
        }
    }

    void update(int x, int v)
    {
        for (; x <= n; x += x & -x)
        {
            bit[x] += v;
        }
    }

    long long get(int x)
    {
        long long res = 0;
        for (; x >= 1; x -= x & -x)
        {
            res += bit[x];
        }
        return res;
    }
};

void solve()
{
    cin >> n >> q;
    for (int i = 1; i <= n; ++i)
    {
        cin >> x[i];
    }
    FenwickTree bit(n);
    for (int i = 1; i <= n; ++i)
    {
        bit.update(i, x[i]);
    }
    while (q--)
    {
        int t;
        cin >> t;
        if (t == 1)
        {
            int k, u;
            cin >> k >> u;
            bit.update(k, u - x[k]);
            x[k] = u;
        }
        else
        {
            int a, b;
            cin >> a >> b;
            cout << bit.get(b) - bit.get(a - 1) << '\n';
        }
    }
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();
    return 0;
}
```

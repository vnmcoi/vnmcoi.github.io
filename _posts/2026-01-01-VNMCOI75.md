Ta vét các hoán vị sử dụng quay lui và mảng đánh dấu.

Hoặc ta có thể sử dụng hàm `next_permutation`.

Code mẫu sử dụng `next_permutaion` (chắc chắn chạy nhanh hơn sử dụng quay lui):

```

#include <bits/stdc++.h>
using namespace std;

const int mxN = 9 + 5;

int N;
int A[mxN];

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> N;
    for (int i = 1; i <= N; ++i)
    {
        A[i] = i;
    }
    do
    {
        for (int i = 1; i <= N; ++i)
        {
            cout << A[i] << ' ';
        }
        cout << '\n';
    } while (next_permutation(A + 1, A + 1 + N));
    return 0;
}
```

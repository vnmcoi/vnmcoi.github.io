Để xác định có bao nhiêu thành phần liên thông ta sẽ kiểm tra bằng hai thuật toán là BFS (thăm theo chiều rộng) và DFS (thăm theo chiều sâu).

Nếu một đỉnh chưa được thăm ta đánh dấu là một thành phần liên thông riêng rồi đánh dấu các đỉnh trong thành phần liên thông đó.

Kiểm tra đầy đủ $N$ đỉnh để đảm bảo tất cả các đỉnh đã được thăm.

```cpp
#include <bits/stdc++.h>
using namespace std;

const int mxN = 2e5 + 5;

int N, M;
vector<int> adj[mxN];
bitset<mxN> vis;

void dfs(int u)
{
    vis[u] = true;
    for (const int &v : adj[u])
    {
        if (vis[v] == false)
        {
            dfs(v);
        }
    }
}

void solve()
{
    cin >> N >> M;
    for (int i = 1; i <= M; ++i)
    {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    int cnt = 0;
    for (int i = 1; i <= N; ++i)
    {
        if (vis[i] == false)
        {
            dfs(i);
            ++cnt;
        }
    }
    cout << cnt;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    solve();
    return 0;
}
```
